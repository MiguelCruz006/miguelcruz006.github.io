GLPI est déployé sur un serveur Debian avec la stack LAMP (Apache, MariaDB, PHP). La base de données est créée et GLPI est configuré via son interface web d'installation.

SRV-GLPI-1 — Installation LAMP + GLPI 10.0.x
# ── Prérequis LAMP ──────────────────────────────────────────────────
root@srv-glpi:~# apt update && apt install -y apache2 mariadb-server \
    php php-mysql php-curl php-gd php-mbstring php-xml php-zip \
    php-intl php-ldap php-apcu php-xmlrpc

# ── Création de la base de données GLPI ─────────────────────────────
root@srv-glpi:~# mysql -u root -p
MariaDB> CREATE DATABASE glpi CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
MariaDB> CREATE USER 'glpi'@'localhost' IDENTIFIED BY 'GlpiP@ss!';
MariaDB> GRANT ALL ON glpi.* TO 'glpi'@'localhost'; FLUSH PRIVILEGES;

# ── Téléchargement et déploiement GLPI ──────────────────────────────
root@srv-glpi:~# wget https://github.com/glpi-project/glpi/releases/download/10.0.15/glpi-10.0.15.tgz
root@srv-glpi:~# tar xzf glpi-10.0.15.tgz -C /var/www/html/
root@srv-glpi:~# chown -R www-data:www-data /var/www/html/glpi
root@srv-glpi:~# systemctl restart apache2
# → Finir l'installation via http://10.2.0.10/glpi (assistant web)

Sauvegarde journalière automatisée + restauration
Cron 2h00 — mysqldump + tar — rotation 7 jours
/usr/local/bin/backup-glpi.sh
#!/bin/bash
# ── Variables ────────────────────────────────────────────────────────
DATE=$(date +%Y%m%d_%H%M)
BACKUP_DIR="/var/backups/glpi"
DB_USER="glpi"  ;  DB_PASS="GlpiP@ss!"  ;  DB_NAME="glpi"
GLPI_DIR="/var/www/html/glpi"

mkdir -p $BACKUP_DIR/$DATE

# ── Sauvegarde base de données MariaDB ──────────────────────────────
mysqldump -u $DB_USER -p$DB_PASS $DB_NAME > $BACKUP_DIR/$DATE/glpi_db.sql

# ── Sauvegarde fichiers GLPI ─────────────────────────────────────────
tar czf $BACKUP_DIR/$DATE/glpi_files.tar.gz $GLPI_DIR/files/

# ── Rotation : suppression des sauvegardes > 7 jours ────────────────
find $BACKUP_DIR -maxdepth 1 -type d -mtime +7 -exec rm -rf {} +

echo "[$(date)] Backup $DATE OK" >> /var/log/backup-glpi.log
Planification cron + test de restauration
# ── Rendre le script exécutable et planifier ────────────────────────
root@srv-glpi:~# chmod +x /usr/local/bin/backup-glpi.sh
root@srv-glpi:~# crontab -e
# Ajouter : 0 2 * * * /usr/local/bin/backup-glpi.sh

# ── Vérifier les sauvegardes présentes ──────────────────────────────
root@srv-glpi:~# ls -lh /var/backups/glpi/

# ── Test de restauration depuis une sauvegarde ───────────────────────
root@srv-glpi:~# mysql -u glpi -p glpi < /var/backups/glpi/20260424_020000/glpi_db.sql
root@srv-glpi:~# tar xzf /var/backups/glpi/20260424_020000/glpi_files.tar.gz -C /

SRV-HAPROXY — /etc/haproxy/haproxy.cfg
root@haproxy:~# apt install -y haproxy

global
    log /dev/log local0
    maxconn 2000

defaults
    mode    http
    timeout connect 5s
    timeout client  30s
    timeout server  30s
    option  forwardfor

frontend glpi_front
    bind            *:443 ssl crt /etc/haproxy/certs/glpi.pem
    redirect        scheme https if !{ ssl_fc }
    default_backend glpi_servers

backend glpi_servers
    balance   roundrobin
    option    httpchk GET /glpi/index.php
    server glpi1 10.2.0.10:80 check inter 2s rise 2 fall 3
    server glpi2 10.2.0.11:80 check inter 2s rise 2 fall 3 backup

listen stats
    bind  *:8404
    stats enable
    stats uri /haproxy-stats
    stats auth admin:admin
check inter 2s rise 2 fall 3 : vérifie toutes les 2s, considère UP après 2 succès consécutifs, DOWN après 3 échecs. Le mot-clé backup sur glpi2 signifie qu'il n'est activé qu'en cas de panne de glpi1.
