Installation et configuration Active Directory + DNS + DHCP
SRV-AD — 10.2.0.101 — Windows Server 2022 — domaine : tierlieux86.lan

Le serveur SRV-AD héberge le contrôleur de domaine principal de la forêt tierlieux86.lan, ainsi que les rôles DNS et DHCP pour la gestion automatique des adresses IP du réseau 10.2.0.0/24.

PowerShell (Admin) — Installation AD DS + DNS + DHCP
# ── 1. Définir l'IP fixe du serveur ────────────────────────────────
PS C:\> New-NetIPAddress -InterfaceAlias "Ethernet" -IPAddress 10.2.0.101 -PrefixLength 24 -DefaultGateway 10.2.0.1
PS C:\> Set-DnsClientServerAddress -InterfaceAlias "Ethernet" -ServerAddresses 127.0.0.1

# ── 2. Installer les rôles AD DS, DNS et DHCP ───────────────────────
PS C:\> Install-WindowsFeature -Name AD-Domain-Services, DNS, DHCP -IncludeManagementTools

# ── 3. Créer la forêt Active Directory ─────────────────────────────
PS C:\> Install-ADDSForest `
    -DomainName "tierlieux86.lan" `
    -DomainNetbiosName "TIERLIEUX86" `
    -ForestMode "WinThreshold" `
    -DomainMode "WinThreshold" `
    -InstallDns:$true `
    -SafeModeAdministratorPassword (ConvertTo-SecureString "P@ssw0rd!" -AsPlainText -Force) `
    -Force:$true
# ── Le serveur redémarre automatiquement après cette commande ───────

# ── 4. Configurer le serveur DHCP (après redémarrage) ───────────────
PS C:\> Add-DhcpServerv4Scope `
    -Name "Réseau PARCINFO" `
    -StartRange 10.2.0.50 `
    -EndRange 10.2.0.200 `
    -SubnetMask 255.255.255.0 `
    -State Active
PS C:\> Set-DhcpServerv4OptionValue -Router 10.2.0.1 -DnsServer 10.2.0.101 -DnsDomain "tierlieux86.lan"
PS C:\> Add-DhcpServerInDC -DnsName "srv-ad.tierlieux86.lan" -IPAddress 10.2.0.101

# ── 5. Ajouter zone DNS inverse ─────────────────────────────────────
PS C:\> Add-DnsServerPrimaryZone `
    -NetworkID "10.2.0.0/24" `
    -ReplicationScope "Forest"
PS C:\> Add-DnsServerResourceRecordA -ZoneName "tierlieux86.lan" -Name "srv-glpi" -IPv4Address "10.2.0.100"
PS C:\> Add-DnsServerResourceRecordA -ZoneName "tierlieux86.lan" -Name "glpi" -IPv4Address "10.2.0.100"

1
Sécurisation HTTPS — Certificat SSL/TLS Apache
SRV-GLPI — 10.2.0.100 — Apache2 + OpenSSL
SRV-GLPI — Activation HTTPS
# ── 1. Générer un certificat SSL auto-signé ──────────────────────────
sio@vpsdebian:~$ sudo openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
    -keyout /etc/ssl/private/glpi.key \
    -out /etc/ssl/certs/glpi.crt \
    -subj "/C=FR/ST=IDF/O=TiersLieux86/CN=glpi.tierlieux86.lan"

# ── 2. Activer les modules Apache SSL + Rewrite ──────────────────────
sio@vpsdebian:~$ sudo a2enmod ssl rewrite

# ── 3. Activer le site SSL ───────────────────────────────────────────
sio@vpsdebian:~$ sudo ln -s /etc/apache2/sites-available/default-ssl.conf /etc/apache2/sites-enabled/

# ── 4. Modifier default-ssl.conf ─────────────────────────────────────
sio@vpsdebian:~$ sudo nano /etc/apache2/sites-available/default-ssl.conf
# Modifier :
<VirtualHost *:443>
    ServerName            glpi.tierlieux86.lan
    DocumentRoot          /var/www/html/glpi
    SSLEngine             on
    SSLCertificateFile    /etc/ssl/certs/glpi.crt
    SSLCertificateKeyFile /etc/ssl/private/glpi.key
</VirtualHost>

# ── 5. Redirection HTTP → HTTPS dans 000-default.conf ───────────────
sio@vpsdebian:~$ sudo nano /etc/apache2/sites-available/000-default.conf
# Ajouter dans <VirtualHost *:80> :
Redirect "/" "https://glpi.tierlieux86.lan/"

# ── 6. Redémarrer Apache ─────────────────────────────────────────────
sio@vpsdebian:~$ sudo systemctl restart apache2

Sauvegarde journalière automatisée + restauration
Cron 2h00 — mysqldump + tar — rotation 7 jours — /var/backups/glpi/
/usr/local/bin/backup-glpi.sh
#!/bin/bash
# ── Variables ────────────────────────────────────────────────────────
DATE=$(date +%Y%m%d_%H%M)
BACKUP_DIR="/var/backups/glpi"
DB_USER="glpi"  ;  DB_PASS="GlpiP@ss!"  ;  DB_NAME="glpi"
GLPI_DIR="/var/www/html/glpi"

mkdir -p $BACKUP_DIR/$DATE

# ── Sauvegarde base de données MariaDB ──────────────────────────────
mysqldump -u $DB_USER -p$DB_PASS $DB_NAME > $BACKUP_DIR/$DATE/glpi_db.sql

# ── Sauvegarde fichiers GLPI ─────────────────────────────────────────
tar czf $BACKUP_DIR/$DATE/glpi_files.tar.gz $GLPI_DIR/files/

# ── Rotation : suppression sauvegardes > 7 jours ───────────────────
find $BACKUP_DIR -maxdepth 1 -type d -mtime +7 -exec rm -rf {} +

echo "[$(date)] Backup $DATE OK" >> /var/log/backup-glpi.log
Planification cron + vérification + restauration
# ── Rendre exécutable et planifier ──────────────────────────────────
sio@vpsdebian:~$ sudo chmod +x /usr/local/bin/backup-glpi.sh
sio@vpsdebian:~$ sudo crontab -e
# Ajouter : 0 2 * * * /usr/local/bin/backup-glpi.sh

# ── Tester manuellement ──────────────────────────────────────────────
sio@vpsdebian:~$ sudo bash /usr/local/bin/backup-glpi.sh
sio@vpsdebian:~$ ls -lh /var/backups/glpi/
# → drwxr-xr-x 2 root root 4,0K 3 juin 20:12 20260603_2012
# → -rw-r--r-- 1 root root 805K 3 juin 20:12 glpi_db.sql

# ── Test de restauration ─────────────────────────────────────────────

sio@vpsdebian:~$ sudo tar xzf /var/backups/glpi/20260603_2012/glpi_files.tar.gz -C /

# ── Installation HaProxy ─────────────────────────────────────────────

# ── Configuration /etc/haproxy/haproxy.cfg ──────────────────────────
sio@vpsdebian:~$ sudo nano /etc/haproxy/haproxy.cfg

frontend glpi_front
    bind            *:443 ssl crt /etc/haproxy/certs/glpi.pem
    redirect        scheme https if !{ ssl_fc }
    default_backend glpi_servers

backend glpi_servers
    balance   roundrobin
    option    httpchk GET /glpi/index.php
    server glpi1 10.2.0.100:80 check inter 2s rise 2 fall 3

    bind  *:8404
    stats enable
    stats uri /haproxy-stats
    stats auth admin:admin
